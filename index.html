<script>

const message = "I don’t know when it started, but somewhere between our conversations and the little moments, you became someone I genuinely look forward to. It’s not just about liking you… it’s about how you make things feel effortless, calm, and real. I respect who you are, and I don’t want to rush anything. I will be there for you anytime and every time. I would also like to know you better.";

let i = 0;


function typeWriter() {

  if (i < message.length) {

    document.getElementById("typedText").innerHTML += message.charAt(i);

    i++;

    setTimeout(typeWriter, 30);

  }

}


function startJourney() {

  document.querySelectorAll("section").forEach(sec => sec.classList.add("show"));

  typeWriter();

  createHearts();

}


function yesResponse() {

  document.getElementById('response').classList.add('show');

  document.getElementById('responseText').innerText = "You just made me the happiest person ❤️";

}


function maybeResponse() {

  document.getElementById('response').classList.add('show');

  document.getElementById('responseText').innerText = "Take your time, I respect your feelings 😊";

}


function createHearts() {

  setInterval(() => {

    let heart = document.createElement("div");

    heart.classList.add("heart");

    heart.innerHTML = "❤️";

    heart.style.left = Math.random() * 100 + "vw";

    heart.style.fontSize = Math.random() * 20 + 10 + "px";

    document.body.appendChild(heart);

    setTimeout(() => heart.remove(), 6000);

  }, 300);

}

</script>
